490 static inline void usb_device_tx_config_ep(uint32_t addr, uint32_t ts_type, uint32_t is_double_fifo, uint32_t ep_maxpkt) {
491  uint16_t reg_val = 0;
492  uint16_t temp = 0;
493 
494  /* configure TX CSR register */
495  reg_val = (1 << USBC_BP_TXCSR_D_MODE);
496  reg_val |= (1 << USBC_BP_TXCSR_D_CLEAR_DATA_TOGGLE);
497  reg_val |= (1 << USBC_BP_TXCSR_D_FLUSH_FIFO);
498  writew(reg_val, USBC_REG_TXCSR(addr));
499 
500  if (is_double_fifo) {
501  writew(reg_val, USBC_REG_TXCSR(addr));
502  }
503 
504  /* configure TX EP maximum packet size */
505  reg_val = readw(USBC_REG_TXMAXP(addr));
506  temp = ep_maxpkt & ((1 << USBC_BP_TXMAXP_PACKET_COUNT) - 1);
507  reg_val &= ~(0x1fff);
508  reg_val |= temp;
509  writew(reg_val, USBC_REG_TXMAXP(addr));
510 
511  /* configure EP transfer type */
512  switch (ts_type) {
513  case USBC_TS_TYPE_ISO:
514  usb_device_tx_iso_ep_enable(addr);
515  break;
516  case USBC_TS_TYPE_INT:
517  usb_device_tx_int_ep_enable(addr);
518  break;
519  case USBC_TS_TYPE_BULK:
520  usb_device_tx_bulk_ep_enable(addr);
521  break;
522  default:
523  usb_device_tx_bulk_ep_enable(addr);
524  break;
525  }
526 }

534  uint16_t ep_csr = 0;

536  /* set auto_set, tx_mode, dma_tx_en, and mode1 bits in TX CSR register */
537  ep_csr = readb(USBC_REG_TXCSR(addr) + 1);
