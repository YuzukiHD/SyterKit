62 static inline void arm32_mmu_enable(const uint32_t dram_base, uint64_t dram_size) {
63  uint32_t mmu_base;
64 
65  /* use dram high 16M */
66  if (dram_size > 2048)
67  dram_size = 2048;
68 
69  uint32_t *mmu_base_addr = (uint32_t *) (dram_base + ((dram_size - 1) << 20));
70  uint32_t *page_table = mmu_base_addr;
71 
72  int i;
73  uint32_t reg;
74 
75  /* the front 1M contain BROM/SRAM */
76 #ifdef CONFIG_CHIP_DCACHE
77  page_table[0] = (3 << 10) | (15 << 5) | (1 << 3) | (1 << 2) | 0x2;
78 #else
79  page_table[0] = (3 << 10) | (15 << 5) | (1 << 3) | (0 << 2) | 0x2;
80 #endif
81  /* the front 1G of memory(treated as 4G for all) is set up as none cacheable */
82  for (i = 1; i < (dram_base >> 20); i++) {
83  page_table[i] = (i << 20) | (3 << 10) | (15 << 5) | (0 << 3) | 0x2;
84  }
85  /* Set up as write back and buffered for other 3GB, rw for everyone */
86  for (i = (dram_base >> 20); i < 4096; i++) {
87 #ifdef CONFIG_CHIP_DCACHE
88  page_table[i] = (i << 20) | (3 << 10) | (15 << 5) | (1 << 3) | (1 << 2) | 0x2;
89 #else
90  page_table[i] = (i << 20) | (3 << 10) | (15 << 5) | (1 << 3) | (0 << 2) | 0x2;
91 #endif
92  }
93  /* flush tlb */
94  asm volatile("mcr p15, 0, %0, c8, c7, 0"
95  :
96  : "r"(0));
97  /* Copy the page table address to cp15 */
98  mmu_base = (uint32_t) mmu_base_addr;
99  mmu_base |= (1 << 0) | (1 << 1) | (2 << 3);
100  asm volatile("mcr p15, 0, %0, c2, c0, 0"
101  :
102  : "r"(mmu_base)
103  : "memory");
104  asm volatile("mcr p15, 0, %0, c2, c0, 1"
105  :
106  : "r"(mmu_base)
107  : "memory");
108  /* Set the access control to all-supervisor */
109  asm volatile("mcr p15, 0, %0, c3, c0, 0"
110  :
111  : "r"(0x55555555));//modified, origin value is (~0)
112  asm volatile("isb");
113 
114 #ifdef CONFIG_CHIP_DCACHE
115  /* enable smp */
116  asm volatile("mrc p15, 0, r0, c1, c0, 1");
117  asm volatile("orr r0, r0, #0x040");
118  asm volatile("mcr p15, 0, r0, c1, c0, 1");
119 #endif
120 
121  /* and enable the mmu */
122  asm volatile("mrc p15, 0, %0, c1, c0, 0 @ get CR"
123  : "=r"(reg)
124  :
125  : "cc");
126 
127  sdelay(100);
128  reg |= ((1 << 0) | (1 << 12));// enable mmu, icache
129  reg &= ~(1 << 2); // disable dcache
130 
131  printk(LOG_LEVEL_TRACE, "MMU: CR = 0x%08x\n", reg);
132  asm volatile("mcr p15, 0, %0, c1, c0, 0 @ set CR"
133  :
134  : "r"(reg)
135  : "cc");
136  asm volatile("isb");
137 }

139 static inline void arm32_mmu_disable(void) {
140  uint32_t reg;
141 
142  /* and disable the mmu */
143  asm volatile("mrc p15, 0, %0, c1, c0, 0 @ get CR"
144  : "=r"(reg)
145  :
146  : "cc");
147  sdelay(100);
148  reg &= ~((7 << 0) | (1 << 12));//disable mmu, icache, dcache
149  asm volatile("mcr p15, 0, %0, c1, c0, 0 @ set CR"
150  :
151  : "r"(reg)
152  : "cc");
153  asm volatile("isb");
154  /*
155  * Invalidate all instruction caches to PoU.
156  * Also flushes branch target cache.
157  */
158  asm volatile("mcr p15, 0, %0, c7, c5, 0"
159  :
160  : "r"(0));
161  /* Invalidate entire branch predictor array */
162  asm volatile("mcr p15, 0, %0, c7, c5, 6"
163  :
164  : "r"(0));
165  /* Full system DSB - make sure that the invalidation is complete */
166  asm volatile("dsb");
167  /* ISB - make sure the instruction stream sees it */
168  asm volatile("isb");
169 }
